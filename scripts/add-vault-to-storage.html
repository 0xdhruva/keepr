Add Vault to LocalStorage

Open browser console and run this:

// Get existing cache
const existingCache = JSON.parse(localStorage.getItem('keepr.vaultCache') || '[]');

// Add your vault
const newVault = {
  vaultPda: 'taNc232vrnmq5NEarKz1imNBuGTXoCZMw1ZMeSXasbX',
  name: 'Test Vault', // Update this with your vault name
  createdAt: Date.now(),
  lastRefreshed: Date.now(),
  unlockUnix: 1727939400, // Update with your unlock time (Oct 3, 12:50 PM IST = 1727939400)
  amountLocked: 1000000, // 1 USDC in lamports
  beneficiary: '9ssrJyXicq9m6FpVeeZuwLbwp3qXoNTgv8u4n31GZphQ', // Update with your beneficiary
  creator: '9ssrJyXicq9m6FpVeeZuwLbwp3qXoNTgv8u4n31GZphQ', // Your wallet
  released: false
};

// Add to cache
existingCache.push(newVault);

// Save back
localStorage.setItem('keepr.vaultCache', JSON.stringify(existingCache));

console.log('Vault added to localStorage!');
console.log('Refresh the page to see it in My Vaults');
